# Host del servidor SMTP
# Gmail: smtp.gmail.com
# Outlook: smtp-mail.outlook.com
# Yahoo: smtp.mail.yahoo.com
EMAIL_HOST=smtp.gmail.com

# Puerto SMTP
# 587 = TLS (recomendado)
# 465 = SSL
EMAIL_PORT=587

# Tu correo electrónico
# Este será el correo que envía las notificaciones
EMAIL_USER=tu-email@gmail.com

# Contraseña de aplicación
# ⚠️ IMPORTANTE: Para Gmail, NO uses tu contraseña normal
# Genera una "Contraseña de aplicación" en:
# https://myaccount.google.com/apppasswords
# 
# Pasos:
# 1. Activa la verificación en dos pasos en tu cuenta de Google
# 2. Ve a "Contraseñas de aplicaciones"
# 3. Selecciona "Correo" y "Otro dispositivo"
# 4. Copia la contraseña de 16 caracteres (sin espacios)
